on mouseUp
  set the myArray["age"] of me to "very old"
end mouseUp

setprop myArray[whichKey] pValue
  switch whichKey
    case "name"
      answer "Your name is" && pValue
      break
    case "age"
      answer "You are" && pValue
      break
  end switch
end myArray
